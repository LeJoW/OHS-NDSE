\def\dayTitleShort{}
\def\officeTitleShort{}

\newdimen\titleSpaceLength
\titleSpaceLength=\fontdimen2\font

\newcommand{\titleSpace}{\hspace{\titleSpaceLength}}

\newenvironment{dayTitle}[2]{
	\pagebreak
	\titleSpaceLength=1.1em
	\thispagestyle{empty}
	\def\dayTitleShort{#2}
	\centering \onehalfspacing \Huge\garamond\MakeUppercase
}{
	%{\centering\it\garamond #1\par}
}

\newenvironment{officeTitle}[1]{
	\titleSpaceLength=0.5em
	\def\officeTitleShort{#1}
	\centering \MakeUppercase\it\garamond
}{}

\newcommand{\sectionTitle}[1]{
	\addvspace{5mm}
	{\centering\MakeUppercase{#1}\par}
	\addvspace{7mm}
}

\newcommand{\chapterTitle}[2]{
	\needspace{5mm}
	\vspace{2mm}
	\def\detail{#2}
	\ifx\empty\detail
		{\centering\garamond #1\par}
		\vspace{2mm}
	\else
		{\hfill\garamond#1\hfill\hfill{\it\garamond #2}\hfill}
	\fi
}

\newenvironment{psalmTitle}{\needspace{5mm}\addvspace{2mm}\centering}{}
