\def\dayTitleShort{}
\def\officeTitleShort{}

\newdimen\titleSpaceLength
\titleSpaceLength=\fontdimen2\font

\newcommand{\titleSpace}{\hspace{\titleSpaceLength}}

\newcommand{\dayTitle}[3]{
	\pagebreak
	\vspace*{4mm}
	{
		\titleSpaceLength=1.1em
		\thispagestyle{empty}
		{\centering \onehalfspacing \Huge\MakeUppercase{\garamond #1}\par}
		\vspace{5mm}
		{\centering\it\garamond #2\par}
	}
	\def\dayTitleShort{#3}
}

\newcommand{\officeTitle}[2]{
\addvspace{5mm}
{
\titleSpaceLength=0.5em
{\centering \MakeUppercase{\it\garamond #1}\par}
}
\addvspace{5mm}
\def\officeTitleShort{#2}
}

\newcommand{\sectionTitle}[1]{
	\addvspace{5mm}
	{\centering\MakeUppercase{#1}\par}
	\addvspace{7mm}
}

\newcommand{\chapterTitle}[2]{
	\needspace{5mm}
	\vspace{2mm}
	\def\detail{#2}
	\ifx\empty\detail
		{\centering\garamond #1\par}
		\vspace{2mm}
	\else
		{\hfill\garamond#1\hfill\hfill{\it\garamond #2}\hfill}
	\fi
}

\newcommand{\psalmTitle}[1]{\needspace{5mm}\addvspace{2mm}{\centering #1\par}}
