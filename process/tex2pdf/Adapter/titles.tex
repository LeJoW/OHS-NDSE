\def\dayTitleShort{}
\def\officeTitleShort{}

\newdimen\titleSpaceLength
\titleSpaceLength=\fontdimen2\font

\newcommand{\titleSpace}{\hspace{\titleSpaceLength}}

\newcommand{\dayTitle}[3]{
\thispagestyle{empty}
\begin{center}
	{\titleSpaceLength=1.1em \onehalfspacing \Huge\MakeUppercase{#1}\par}

	{\vspace{1em}}
	{\it #2}
\end{center}
{\addvspace{1em}}
\def\dayTitleShort{#3}
}

\newcommand{\officeTitle}[1]{
{\addvspace{1.5em}}
\begin{center}
	{\it\MakeUppercase{#1}}
\end{center}
{\addvspace{1em}}
\def\officeTitleShort{#1}
}

\newcommand{\sectionTitle}[1]{
	\begin{center}
		{\MakeUppercase{#1}}
	\end{center}
}

\newcommand{\chapterTitle}[2]{
	\vspace{1em}
	\needspace{2em}
	\def\detail{#2}
	\ifx\empty\detail
		\begin{center}
			#1
		\end{center}
	\else
		\hfill#1\hfill\hfill{\it #2}\hfill
	\fi
}
