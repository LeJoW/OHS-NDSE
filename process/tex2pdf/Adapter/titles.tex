\def\dayTitleShort{}
\def\officeTitleShort{}

\newdimen\titleSpaceLength
\titleSpaceLength=\fontdimen2\font

\newcommand{\titleSpace}{\hspace{\titleSpaceLength}}

\newcommand{\dayTitle}[3]{
\thispagestyle{empty}
\begin{center}
	{\titleSpaceLength=1.1em \onehalfspacing \Huge\MakeUppercase{\garamond #1}\par}

	{\vspace{1em}}
	{\it\garamond #2}
\end{center}
{\addvspace{1em}}
\def\dayTitleShort{#3}
}

\newcommand{\officeTitle}[2]{
\titleSpaceLength=0.5em
{\addvspace{1.5em}}
\begin{center}
	{\MakeUppercase{\it\garamond #1}}
\end{center}
{\addvspace{1em}}
\def\officeTitleShort{#2}
}

\newcommand{\sectionTitle}[1]{
	\begin{center}
		{\MakeUppercase{#1}}
	\end{center}
}

\newcommand{\chapterTitle}[2]{
	\vspace{1em}
	\needspace{2em}
	\def\detail{#2}
	\ifx\empty\detail
		\begin{center}
			\garamond #1
		\end{center}
	\else
		{\hfill\garamond#1\hfill\hfill{\it\garamond #2}\hfill}
	\fi
}
