\usepackage{pdfpages}

\newcommand*{\pdfnumberofpages}[1]{%
	\directlua{%
		local doc = pdfe.open("\luaescapestring{#1}")
		local pages
		if (doc) then
		pages = pdfe.getnofpages(doc)
		else
		pages = 0
		end
		tex.write(pages)
	}%
}

\newcounter{gabcCount}
\newcommand{\gabc}[1]{
	\setcounter{gabcCount}{0}
	\setlength{\parindent}{0pt}
	\setlength{\parskip}{0pt}
	\whiledo{\value{gabcCount}<\pdfnumberofpages{#1}}{
		\stepcounter{gabcCount}
		\includegraphics[page=\thegabcCount,width=1\linewidth]{#1}

	}
}
